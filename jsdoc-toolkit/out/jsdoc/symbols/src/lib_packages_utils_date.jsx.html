1 /**

  4  * @function

  6  */
  7 Date.prototype.year_month_day = function() {
  8     var _year = this.getFullYear();
  9     var _month = this.getMonth() + 1;
 10     if (_month.toString().length == 1) _month = "0" +_month;
 11     var _day = this.getDate();
