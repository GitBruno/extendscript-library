/**
 * Function to check if the same file is used many times (image, PDF)
 * @name Document.is_link_used_many_time
 * @function
 * @param {Link} link_item the link item
 * @return {Number} The number of time that the link is used
 */
Document.prototype.is_link_used_many_time = function(link_item){
    var array_all_links = this.links;
    var counter = 0;
    for(var i = 0; i < array_all_links.length; i++){
        var current_link = array_all_links[i];
        if(link_item.filePath === current_link.filePath){
            counter++;
        }
    }
    return counter;
}
