PS.Constants = (function (my) {

    /**
     * Object to set the constants
     * @private
     * @type {{MY_CONST: string, MY_INTEGER: number}}
     */
    var MY_CONSTANTS = {

        'MY_CONST': 'hello constante',
        'MY_INTEGER': 1
    };

    /**
     * Function to get a constant
     * @name Lib.Constants.get
     * @param {String } name the name of the constant
     * @returns {*}
     */
    my.get = function (name) {
        return MY_CONSTANTS[name];
    }
    return my;
})(PS.Constants || {});
